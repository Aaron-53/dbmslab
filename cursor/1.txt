

Declare
Cursor c is select Name from staff order by Salary desc;
Names STAFF.Name%type;
Begin
Open c;
For i in 1..6 loop
Fetch c into Names;
Exit when c%notfound;
dbms_output.put_line(Names);
End loop;
Close c;
End;


DECLARE
  CURSOR c IS SELECT Name FROM staff ORDER BY Salary DESC;
  Names STAFF.Name%TYPE;
BEGIN
  OPEN c;
  FETCH c INTO Names;

  WHILE c%FOUND LOOP
    DBMS_OUTPUT.PUT_LINE(Names);
    FETCH c INTO Names;
  END LOOP;

  CLOSE c;
END;
/
