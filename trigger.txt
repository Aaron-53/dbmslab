Create or replace trigger triger1
Before insert or update or delete on customers
FOR EACH ROW
WHEN (NEW.salary>0)
Declare
 Salary_diff number;
Begin
 Salary_diff:=(:NEW.salary-:OLD.salary);
 dbms_output.put_line(Salary_diff);
End;
/
