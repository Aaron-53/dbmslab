Armstrong numbers

Declare 
 a integer:=&a;
temp integer;
digit integer;
sum integer:=0;  //use integer
Begin
 temp:=a;
 While temp>0 loop
  digit:=mod(temp,10);
  sum:=sum+POWER(digit,3);
  temp := trunc(temp/10); //truncate
End loop;
if(sum=a) then
dbms_output.put_line('Arm'); //use these quotes not smart quotes
End if;
end;/
